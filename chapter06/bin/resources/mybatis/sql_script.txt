/* 존재하는 가게 정보 테이블 삭제 */
DROP TABLE SHOP;

/* 가게 정보 테이블 생성 */
CREATE TABLE SHOP (
    SHOP_NO NUMBER NOT NULL ENABLE,
    SHOP_NAME VARCHAR2(100 BYTE),
    SHOP_LOCATION CLOB,
    SHOP_STATUS VARCHAR2(20 BYTE),
    CONSTRAINT SHOP_PK PRIMARY KEY (SHOP_NO)
);
